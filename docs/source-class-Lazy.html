110     * Moves the cursor to the next element (the one after the current element).
111     */
112     public function next() {
113         $this->started_iteration = true;
114         if (!is_array($this->iterable)) {
115             if (!$this->using_cache) {
116                 $this->cache[] = $this->iterable->current();
117             }
118             $this->iterable->next();
119         }
120         else {
121             next($this->iterable);
122             $this->array_pos++;
