 40     /**
 41      * @internal
 42     */
 43     protected function _get_start_end_from_offset($offset) {
 44         if (is_array($offset)) {
 45             if (count($offset) === 1) {
 46                 $offset[] = null;
 47             }
 48             if (is_int($offset[0]) && is_int($offset[1])) {
 49                 $use_slicing = true;
 50                 $start = $offset[0];
 51                 $end = $offset[1];
 52             }
 53             else {
 54                 throw new \Exception('Invalid array offset '.__toString($offset).'. Array offsets must have the form \'[int1(,int2)]\'.');
 55             }
 56         }
 57         else if (is_string($offset)) {
 58             $offset = preg_replace('/\s+/', '', $offset);
 59             if (preg_match('/^\-?\d+\:(\-?\d+)?$/', $offset) === 1) {
 60                 $use_slicing = true;
 61                 $parts = explode(':', $offset);
 62                 $start = (int) $parts[0];
 63                 if (strlen($parts[1]) > 0) {
 64                     $end = (int) $parts[1];
 65                 }
 66                 else {
 67                     $end = null;
 68                 }
 69             }
 70             else {
 71                 throw new \Exception('Invalid string offset \''.$offset.'\'. String offsets must have the form \'int1:(int2)\'.');
 72             }
 73         }
 74         else if (is_int($offset)) {
 75             $use_slicing = false;
 76             $start = $offset;
 77             $end = $offset;
 78         }
 79         else {
 80             throw new \Exception('Invalid offset. Use null ($a[]=4) to push, int for index access or for slicing use [start, end] or \'start:end\'!');
 81         }
 82         try {
 83             $end = $this->_adjust_offset($end);
 84         } catch (\Exception $e) {
 85             $end = $this->size();
 86         }
 87 
 88         return [
 89             'start' => $this->_adjust_offset($start),
 90             'end' => $end,
 91             'slicing' => $use_slicing
 92         ];
 93     }
