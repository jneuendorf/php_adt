480     * This method will internally use multiple calls of str_replace. For that reason using e.g. '{0}'.format('{1}', 2) will result in '2';

486         $last_arg = $args[count($args) - 1];
487         unset($args[count($args) - 1]);

505         for ($i = 0; $i < $new_res_len - 1; $i++) {
506             // append char
507             if (!($new_res[$i] === '{' && $new_res[$i + 1] === '}')) {
508                 $res .= $new_res[$i];
